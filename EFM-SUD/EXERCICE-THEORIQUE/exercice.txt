import {useState, useEffect} from "react" ;

export default function App(){
    const [articles, setArticles] = useState([]) ;
    useEffecte(() => {
        fetch('https://freejson.com/api/articles')
        .then((res) => res.json())
        .then((data) => setArticles(data))
    }, [])
}
